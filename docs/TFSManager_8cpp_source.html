481 std::string TFSManager::calculateHash(std::string path)
482 {
483  unsigned char md5Value[MD5_DIGEST_LENGTH];
484  unsigned char buf[4096];
485  int fd = open(path.c_str(), O_RDONLY);
486  int bytesRead = 0;
487 
488  MD5_CTX md5Context;
489  MD5_Init(&md5Context);
490  while ((bytesRead = read(fd, buf, 4096)) > 0)
491  {
492  MD5_Update(&md5Context, buf, bytesRead);
493  }
494  MD5_Final(md5Value, &md5Context);
495 
496  char md5String[2*MD5_DIGEST_LENGTH + 1];
497  std::string hexSymbols = "0123456789ABCDEF";
498  for (auto i = 0, j = 0; i < 2*MD5_DIGEST_LENGTH; i += 2, j++)
499  {
500  md5String[i] = hexSymbols[md5Value[j] >> 4];
501  md5String[i+1] = hexSymbols[md5Value[j] & 0xF];
502  }
503  md5String[2*MD5_DIGEST_LENGTH] = '\0';
504  return std::string(md5String);
505 }

552 bool TFSManager::dispatch(Message m)
553 {
554  static long loops = 0;
555  loops++;
556  std::vector<std::string> tokens = splitAtFirstOccurance(m.content);
557  std::string query = tokens[0];
558  if (query == "QH_TEST")
559  {
560  messageQueryHandler("TM_ACK (messages dispatched: " + std::to_string(loops) + ")");
561  }
562  else if (query == "QH_EXIT" || query == "FD_EXIT")
563  {
564  return false;
565  }
566  else if (query == "FD_TEST" || query == "FD_LOG")
567  {
568  messageFUSEFileSystem("TM_ACK");
569  }
570  else if (query == "FD_GET_PATH" || query == "FD_GET_PATH_WRITE")
571  {
572  std::string realPath = "";
573  if (tagView == false)
574  {
575  realPath = getFilePath(tokens[1]);
576  }
577  else if (query != "FD_GET_PATH_WRITE")
578  {
579  realPath = getTaggedFilePath(tokens[1]);
580  }
581  messageFUSEFileSystem(realPath);
582  }
583  else if (query == "FD_IF_DIR")
584  {
585  bool isDirectory;
586  if (tagView == true)
587  {
588  isDirectory = (getTagID(tokens[1]) != "");
589  }
590  else
591  {
592  std::vector<std::string> parts = splitPathIntoParts(tokens[1]);
593  isDirectory = (getFolderID(parts) != "");
594  }
595  messageFUSEFileSystem(isDirectory ? "TM_TRUE" : "TM_FALSE");
596  }
597  else if (query == "FD_READ_DIR")
598  {
599  std::vector<std::string> contents;
600  if (tagView == false)
601  {
602  contents = listFolder(tokens[1]);
603  }
604  else
605  {
606  contents = listTagChildren(tokens[1]);
607  }
608  std::size_t size = contents.size();
609  if (size == 0)
610  {
611  messageFUSEFileSystem("");
612  }
613  else
614  {
615  for (std::size_t i = 0; i < size; i++)
616  {
617  bool complete = (i == size - 1);
618  messageFUSEFileSystem(contents[i], complete);
619  }
620  }
621  }
622  else if (query == "FD_MKDIR")
623  {
624  int returnValue = 1;
625  if (tagView == false)
626  {
627  returnValue = createFolder(tokens[1]);
628  }
629  else
630  {
631  returnValue = createTag(tokens[1]);
632  }
633  if (returnValue == 0)
634  {
635  messageFUSEFileSystem("TM_ACK");
636  }
637  else // folder/tag wasn't created
638  {
639  messageFUSEFileSystem(std::to_string(returnValue));
640  }
641  }
642  else if (query == "FD_RMDIR")
643  {
644  int returnValue = 1;
645  if (tagView == false)
646  {
647  returnValue = deleteFolder(tokens[1]);
648  }
649  else
650  {
651  returnValue = deleteTag(tokens[1]);
652  }
653  if (returnValue == 0)
654  {
655  messageFUSEFileSystem("TM_ACK");
656  }
657  else // folder/tag doesn't exist or wasn't deleted
658  {
659  messageFUSEFileSystem(std::to_string(returnValue));
660  }
661  }
662  else if (query == "FD_UNLINK")
663  {
664  int returnValue = 1;
665  if (tagView == false)
666  {
667  returnValue = deleteFile(tokens[1]);
668  }
669  else // untag file, not delete
670  {
671  std::string tagID = getParentTagIDFromPath(tokens[1]);
672  std::string fileID = getTaggedFileID(tagID, getFilename(tokens[1]));
673  returnValue = untagSingleFile(fileID, tagID);
674  }
675  if (returnValue == 0)
676  {
677  messageFUSEFileSystem("TM_ACK");
678  }
679  else // file doesn't exist or wasn't deleted
680  {
681  messageFUSEFileSystem(std::to_string(returnValue));
682  }
683  }
684  else if (query == "FD_RENAME")
685  {
686  int returnValue = 1;
687  std::vector<std::string> arguments = splitAtFirstOccurance(tokens[1], ',');
688  if (tagView == false)
689  {
690  returnValue = renamePath(arguments[0], arguments[1]);
691  }
692  else
693  {
694  returnValue = renameTaggedPath(arguments[0], arguments[1]);
695  }
696  // non-zero return value -> file doesn't exist or wasn't renamed
697  messageFUSEFileSystem((returnValue == 0) ? "TM_ACK" : "TM_FAIL");
698  }
699  else if (query == "FD_TRUNCATE")
700  {
701  int returnValue = 1;
702  std::vector<std::string> arguments = splitAtFirstOccurance(tokens[1], ',');
703  if (tagView == false) // tag view mode read only
704  {
705  off_t length = std::stol(arguments[0]);
706  std::string filePath = arguments[1];
707  returnValue = truncateFile(length, filePath);
708  }
709  if (returnValue == 0)
710  {
711  messageFUSEFileSystem("TM_ACK");
712  }
713  else // file doesn't exist or wasn't truncated
714  {
715  messageFUSEFileSystem(std::to_string(returnValue));
716  }
717  }
718  else if (query == "FD_UPDATE")
719  {
720  if (tagView == false) // tag view mode read only
721  {
722  updateFile(tokens[1]);
723  }
724  messageFUSEFileSystem("TM_ACK");
725  }
726  else if (query == "FD_ADD_TEMP") // in tag view mode, mknod will fail before sending this
727  {
728  addTemporaryFile(tokens[1]);
729  messageFUSEFileSystem("TM_ACK");
730  }
731  else if (query == "QH_TAG")
732  {
733  std::vector<std::string> arguments = splitAtFirstOccurance(tokens[1], ',');
734  int returnValue = tagFiles(arguments[0], arguments[1]);
735  if (returnValue == 0)
736  {
737  messageQueryHandler("File(s) successfully tagged.");
738  }
739  else if (returnValue == EEXIST)
740  {
741  messageQueryHandler("Failed. Filename conflict with "
742  "files already tagged with the same tag.");
743  }
744  else
745  {
746  messageQueryHandler("Failed. Either file(s) path or tag is invalid.");
747  }
748  }
749  else if (query == "QH_UNTAG")
750  {
751  std::vector<std::string> arguments = splitAtFirstOccurance(tokens[1], ',');
752  int returnValue = untagFiles(arguments[0], arguments[1]);
753  if (returnValue == 0)
754  {
755  messageQueryHandler("File(s) successfully untagged.");
756  }
757  else
758  {
759  messageQueryHandler("Failed. Either file(s) path or tag is invalid.");
760  }
761  }
762  else if (query == "QH_NEST")
