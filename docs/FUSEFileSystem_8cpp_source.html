133 std::vector<std::string> FUSEFileSystem::queryTFS(std::string query)
134 {
135  serializeMessage(query.c_str(), buffer);
136  mq_send(txMQ, buffer, TFS_MQ_MESSAGE_SIZE, 0);
137 
138  Message m;
139  std::vector<std::string> results;
140  do
141  {
142  mq_receive(rxMQ, buffer, TFS_MQ_MESSAGE_SIZE, NULL);
143  m = deserializeMessage(buffer);
144  results.push_back(std::string(m.content));
145  } while (m.complete == false);
